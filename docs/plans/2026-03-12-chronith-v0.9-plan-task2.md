## Step 2.8: Create ChronithClient

Create `packages/sdk-typescript/src/client.ts`:

```ts
import { client as heyApiClient } from "./generated/index.js";

export interface ChronithClientOptions {
  baseUrl: string;
  onTokenRefresh?: () => Promise<string | null>;
}

export class ChronithClient {
  readonly correlationId: string;
  private _token: string | null = null;
  private _apiKey: string | null = null;
  private readonly _options: ChronithClientOptions;

  constructor(options: ChronithClientOptions) {
    if (!options.baseUrl) throw new Error("baseUrl is required");
    this._options = options;
    this.correlationId = crypto.randomUUID();

    // Configure the hey-api generated client
    heyApiClient.setConfig({
      baseUrl: options.baseUrl,
      headers: () => this._buildHeaders(),
    });
  }

  setToken(token: string): void {
    this._token = token;
    this._apiKey = null;
  }

  setApiKey(key: string): void {
    this._apiKey = key;
    this._token = null;
  }

  clearAuth(): void {
    this._token = null;
    this._apiKey = null;
  }

  getAuthHeader(): string | null {
    if (this._token) return `Bearer ${this._token}`;
    if (this._apiKey) return `ApiKey ${this._apiKey}`;
    return null;
  }

  private _buildHeaders(): Record<string, string> {
    const headers: Record<string, string> = {
      "X-Correlation-Id": this.correlationId,
    };
    const auth = this.getAuthHeader();
    if (auth) headers["Authorization"] = auth;
    return headers;
  }
}
```

## Step 2.9: Create errors.ts

Create `packages/sdk-typescript/src/errors.ts`:

```ts
/** RFC 7807 Problem Details */
export interface ProblemDetails {
  type?: string;
  title: string;
  status: number;
  detail?: string;
  errors?: Record<string, string[]>;
}

export class ChronithApiError extends Error {
  readonly status: number;
  readonly title: string;
  readonly detail: string | undefined;
  readonly errors: Record<string, string[]> | undefined;

  constructor(problem: ProblemDetails) {
    super(problem.title);
    this.name = "ChronithApiError";
    this.status = problem.status;
    this.title = problem.title;
    this.detail = problem.detail;
    this.errors = problem.errors;
  }
}

export function parseProblemDetails(body: unknown): ChronithApiError | null {
  if (
    typeof body === "object" &&
    body !== null &&
    "status" in body &&
    "title" in body
  ) {
    return new ChronithApiError(body as ProblemDetails);
  }
  return null;
}
```
