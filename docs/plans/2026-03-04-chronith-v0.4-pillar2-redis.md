# Chronith v0.4 — Pillar 2: Redis Infra Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Add Redis-backed rate limiting (`RedisRateLimitStore`) and Redis caching for metrics + availability endpoints, with graceful in-memory fallback when Redis is disabled.

**Architecture:** New `RedisRateLimitStore` drops in behind the existing `IRateLimitStore` abstraction. New `IRedisCacheService` wraps `IDistributedCache` (StackExchange.Redis provider) with JSON serialization; injected into `GetTenantMetricsQueryHandler` and `GetAvailabilityHandler`. Cache is invalidated on booking state changes via a new `CacheInvalidationHandler` that subscribes to `BookingStatusChangedNotification`. DI registration switches between Redis and in-memory based on `Redis:Enabled` config flag. Docker Compose gains a `redis:8-alpine` service.

**Tech Stack:** StackExchange.Redis 2.x, Microsoft.Extensions.Caching.StackExchangeRedis 10.x, Testcontainers.Redis 4.x, NSubstitute, FluentAssertions, .NET 10 `System.Text.Json`

```csharp
using System.Text.Json;
using Chronith.Application.Interfaces;
using Microsoft.Extensions.Caching.Distributed;

namespace Chronith.Infrastructure.Caching;

public sealed class RedisCacheService(IDistributedCache cache) : IRedisCacheService
{
    private static readonly JsonSerializerOptions JsonOptions = new(JsonSerializerDefaults.Web);

    public async Task<T?> GetOrSetAsync<T>(
        string key,
        Func<Task<T>> factory,
        TimeSpan ttl,
        CancellationToken ct = default)
    {
        var bytes = await cache.GetAsync(key, ct);
        if (bytes is not null)
            return JsonSerializer.Deserialize<T>(bytes, JsonOptions);

        var value = await factory();
        if (value is not null)
        {
            var serialized = JsonSerializer.SerializeToUtf8Bytes(value, JsonOptions);
            await cache.SetAsync(key, serialized,
                new DistributedCacheEntryOptions
                {
                    AbsoluteExpirationRelativeToNow = ttl
                }, ct);
        }

        return value;
    }

    public Task InvalidateAsync(string key, CancellationToken ct = default)
        => cache.RemoveAsync(key, ct);
}
```

**Background:** `IRateLimitStore.GetPermitLimit(string tenantId)` returns only the configured permit limit — it does NOT perform counter increment or sliding window logic. The actual counting is done by ASP.NET Core's built-in `RateLimiter` middleware (in `Program.cs`) which calls `GetPermitLimit` to resolve per-tenant limits. So `RedisRateLimitStore` has exactly the same contract as `InMemoryRateLimitStore`: read config, return the limit for a tenant. No Lua scripts or INCR calls needed here.

```csharp
using Chronith.Application.Interfaces;
using Chronith.Application.Options;
using Microsoft.Extensions.Options;

namespace Chronith.Infrastructure.RateLimiting;

/// <summary>
/// Redis-backed IRateLimitStore. Returns the configured per-tenant permit limit,
/// delegating counter storage to ASP.NET Core's built-in RateLimiter middleware.
/// Registered in DI only when Redis:Enabled = true; InMemoryRateLimitStore
/// is used as the fallback otherwise.
/// </summary>
public sealed class RedisRateLimitStore(IOptions<RateLimitingOptions> options) : IRateLimitStore
{
    private readonly RateLimitingOptions _options = options.Value;

    public int GetPermitLimit(string tenantId)
    {
        if (_options.TenantOverrides.TryGetValue(tenantId, out var @override))
            return @override.PermitLimit;

        return _options.DefaultPermitLimit;
    }
}
```

```csharp
namespace Chronith.Application.Queries.Tenant.GetTenantMetrics;

public sealed class GetTenantMetricsQueryHandler(
    ITenantContext tenantContext,
    IBookingRepository bookingRepository,
    IWebhookOutboxRepository outboxRepository,
    IBookingTypeRepository bookingTypeRepository,

        var cacheKey = $"metrics:{tenantId}";

    private async Task<TenantMetricsDto> FetchMetricsAsync(Guid tenantId, CancellationToken cancellationToken)

            ?? throw new NotFoundException(nameof(Tenant), tenantId);

        var tz = TimeZoneInfo.FindSystemTimeZoneById(tenant.TimeZoneId);

            tenantNow.Year, tenantNow.Month, 1, 0, 0, 0, tenantNow.Offset);
        var monthStartUtc = monthStart.ToUniversalTime();

        var types = await bookingTypeRepository.GetTypeMetricsAsync(tenantId, cancellationToken);

            ? Math.Round((decimal)deliveries.Delivered / totalDeliveries * 100, 1)
            : null;

            Bookings: new BookingMetricsDto(
                bookings.Total,

                    kvp => kvp.Key.ToString().ToSnakeCase(),
                    kvp => kvp.Value),
                bookings.ThisMonth),
            Webhooks: new WebhookMetricsDto(
                totalDeliveries, deliveries.Delivered, deliveries.Failed, deliveryRatePct),
            BookingTypes: new BookingTypeMetricsDto(types.Active, types.Archived));
    }
```

```csharp
using Chronith.Application.DTOs;
using Chronith.Application.Interfaces;
using Chronith.Domain.Enums;
using Chronith.Domain.Exceptions;
using MediatR;

namespace Chronith.Application.Queries.Availability;

// ── Query ─────────────────────────────────────────────────────────────────────

public sealed record GetAvailabilityQuery : IRequest<AvailabilityDto>, IQuery
{
    public required string BookingTypeSlug { get; init; }
    public required DateTimeOffset From { get; init; }
    public required DateTimeOffset To { get; init; }
}

// ── Handler ───────────────────────────────────────────────────────────────────

public sealed class GetAvailabilityHandler(
    ITenantContext tenantContext,
    IBookingTypeRepository bookingTypeRepo,
    IBookingRepository bookingRepo,
    ITenantRepository tenantRepo,
    ISlotGeneratorService slotGenerator,
    IRedisCacheService? cacheService = null)
    : IRequestHandler<GetAvailabilityQuery, AvailabilityDto>
{
    private static readonly BookingStatus[] ConflictStatuses =
    [
        BookingStatus.PendingPayment,
        BookingStatus.PendingVerification,
        BookingStatus.Confirmed
    ];

    public Task<AvailabilityDto> Handle(GetAvailabilityQuery query, CancellationToken ct)
    {
        if (cacheService is not null)
        {
            // Key includes booking type ID (loaded below) and date of the From parameter.
            // We load the booking type first to get its Id, then delegate to cache.
            return HandleWithCacheAsync(query, ct);
        }

        return FetchAvailabilityAsync(query, ct);
    }

    private async Task<AvailabilityDto> HandleWithCacheAsync(GetAvailabilityQuery query, CancellationToken ct)
    {
        var bookingType = await bookingTypeRepo.GetBySlugAsync(tenantContext.TenantId, query.BookingTypeSlug, ct)
            ?? throw new NotFoundException("BookingType", query.BookingTypeSlug);

        var cacheKey = $"availability:{bookingType.Id}:{query.From.UtcDateTime:yyyy-MM-dd}";

        return (await cacheService!.GetOrSetAsync<AvailabilityDto>(
            cacheKey,
            () => FetchAvailabilityInternalAsync(bookingType, query, ct),
            TimeSpan.FromSeconds(30),
            ct))!;
    }

    private async Task<AvailabilityDto> FetchAvailabilityAsync(GetAvailabilityQuery query, CancellationToken ct)
    {
        var bookingType = await bookingTypeRepo.GetBySlugAsync(tenantContext.TenantId, query.BookingTypeSlug, ct)
            ?? throw new NotFoundException("BookingType", query.BookingTypeSlug);

        return await FetchAvailabilityInternalAsync(bookingType, query, ct);
    }

    private async Task<AvailabilityDto> FetchAvailabilityInternalAsync(
        Domain.Models.BookingType bookingType,
        GetAvailabilityQuery query,
        CancellationToken ct)
    {
        var tenant = await tenantRepo.GetByIdAsync(tenantContext.TenantId, ct)
            ?? throw new NotFoundException("Tenant", tenantContext.TenantId);

        var tz = tenant.GetTimeZone();

        var bookedSlots = await bookingRepo.GetBookedSlotsAsync(
            bookingType.Id, query.From, query.To, ConflictStatuses, ct);

        var slots = slotGenerator.GenerateAvailableSlots(
            bookingType, tz, query.From, query.To, bookedSlots);

        return new AvailabilityDto(
            slots.Select(s => new AvailableSlotDto(s.Start, s.End)).ToList());
    }
}
```
