## Step 9.1: Create the dashboard CI workflow

Create `.github/workflows/dashboard-ci.yml`:

```yaml
name: Dashboard CI

on:
  push:
    branches: [main, develop, "feat/**"]
    paths:
      - "dashboard/**"
      - ".github/workflows/dashboard-ci.yml"
  pull_request:
    branches: [main, develop]
    paths:
      - "dashboard/**"
      - ".github/workflows/dashboard-ci.yml"

defaults:
  run:
    working-directory: dashboard

jobs:
  lint-typecheck:
    name: Lint + Type Check
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-node@v4
        with:
          node-version: "22"
          cache: "npm"
          cache-dependency-path: dashboard/package-lock.json

      - name: Install dependencies
        run: npm ci

      - name: Lint
        run: npm run lint

      - name: Type check
        run: npx tsc --noEmit

  unit-tests:
    name: Unit Tests (Vitest)
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-node@v4
        with:
          node-version: "22"
          cache: "npm"
          cache-dependency-path: dashboard/package-lock.json

      - name: Install dependencies
        run: npm ci

      - name: Run Vitest
        run: npx vitest run --reporter=verbose

  build:
    name: Next.js Build
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-node@v4
        with:
          node-version: "22"
          cache: "npm"
          cache-dependency-path: dashboard/package-lock.json

      - name: Install dependencies
        run: npm ci

      - name: Build
        run: npm run build
        env:
          CHRONITH_API_URL: http://localhost:5001

  docker-build:
    name: Docker Build
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: Set up Docker Buildx
        uses: docker/setup-buildx-action@v3

      - name: Build dashboard image
        uses: docker/build-push-action@v6
        with:
          context: ./dashboard
          file: ./dashboard/Dockerfile
          push: false
          tags: chronith-dashboard:ci
          cache-from: type=gha
          cache-to: type=gha,mode=max
```

## Step 9.3: Create the dashboard Dockerfile

Create `dashboard/Dockerfile`:

```dockerfile
# Stage 1: deps
FROM node:22-alpine AS deps
WORKDIR /app
COPY package.json package-lock.json ./
RUN npm ci

# Stage 2: builder
FROM node:22-alpine AS builder
WORKDIR /app
COPY --from=deps /app/node_modules ./node_modules
COPY . .
ENV NEXT_TELEMETRY_DISABLED=1
RUN npm run build

# Stage 3: runner
FROM node:22-alpine AS runner
WORKDIR /app
ENV NODE_ENV=production
ENV NEXT_TELEMETRY_DISABLED=1

RUN addgroup --system --gid 1001 nodejs
RUN adduser --system --uid 1001 nextjs

COPY --from=builder /app/public ./public
COPY --from=builder --chown=nextjs:nodejs /app/.next/standalone ./
COPY --from=builder --chown=nextjs:nodejs /app/.next/static ./.next/static

USER nextjs
EXPOSE 3000
ENV PORT=3000
ENV HOSTNAME="0.0.0.0"

CMD ["node", "server.js"]
```

## Step 9.4: Update docker-compose.yml to include the dashboard

Read the existing `docker-compose.yml` first, then add the dashboard service.

Open `docker-compose.yml` and append the dashboard service to the `services` block:

```yaml
dashboard:
  build:
    context: ./dashboard
    dockerfile: Dockerfile
  ports:
    - "3000:3000"
  environment:
    - CHRONITH_API_URL=http://api:5001
    - NODE_ENV=production
  depends_on:
    - api
  profiles:
    - dashboard
```

The `profiles: [dashboard]` key means `docker compose up` (without `--profile dashboard`) will not start the dashboard by default. To start with dashboard:

```bash
docker compose --profile dashboard up -d
```

This avoids breaking the existing CI pipeline which doesn't build the dashboard image.
