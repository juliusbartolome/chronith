### Task 2: Fix SanitizationBehavior — init-only backing field reflection

**Files:**
- Modify: `src/Chronith.Application/Behaviors/SanitizationBehavior.cs`
- Modify: `tests/Chronith.Tests.Unit/Behaviors/SanitizationBehaviorTests.cs`

**Context:**
`SanitizationBehavior<TRequest,TResponse>` is a MediatR pipeline behavior that strips HTML from string properties. It currently skips any property where `IsInitOnly(p)` returns true — which is every property on every `record` command (they all use `required init`). **Step 3: Implement fix in SanitizationBehavior**

In `Handle`, change the property iteration so that after computing `sanitized`, if the property is init-only, look up the backing field and set it:

```csharp
public async Task<TResponse> Handle(
    TRequest request,
    RequestHandlerDelegate<TResponse> next,
    CancellationToken cancellationToken)
{
    var properties = typeof(TRequest)
        .GetProperties(BindingFlags.Public | BindingFlags.Instance)
        .Where(p => p.PropertyType == typeof(string));

    foreach (var property in properties)
    {
        var value = property.GetValue(request) as string;
        if (value is null)
            continue;

        var sanitized = DangerousBlockRegex().Replace(value, string.Empty);
        sanitized = HtmlTagRegex().Replace(sanitized, string.Empty);

        if (!IsInitOnly(property))
        {
            property.SetValue(request, sanitized);
        }
        else
        {
            // For init-only properties (record types), set the backing field directly
            var backingField = typeof(TRequest).GetField(
                $"<{property.Name}>k__BackingField",
                BindingFlags.NonPublic | BindingFlags.Instance);
            backingField?.SetValue(request, sanitized);
        }
    }

    return await next(cancellationToken);
}
```

Remove the `.Where(p => ... && !IsInitOnly(p))` from the LINQ — the init-only check moves inside the loop.

```csharp
internal async Task GenerateBookingsAsync(CancellationToken ct)
{
    using var scope = scopeFactory.CreateScope();
    var recurrenceRuleRepo = scope.ServiceProvider.GetRequiredService<IRecurrenceRuleRepository>();
    var bookingTypeRepo = scope.ServiceProvider.GetRequiredService<IBookingTypeRepository>();
    var tenantRepo = scope.ServiceProvider.GetRequiredService<ITenantRepository>();
    var customerRepo = scope.ServiceProvider.GetRequiredService<ICustomerRepository>();
    var bookingRepo = scope.ServiceProvider.GetRequiredService<IBookingRepository>();
    var unitOfWork = scope.ServiceProvider.GetRequiredService<IUnitOfWork>();
    var publisher = scope.ServiceProvider.GetRequiredService<IPublisher>();

    var rules = await recurrenceRuleRepo.GetAllActiveAcrossTenantsAsync(ct);
    if (rules.Count == 0) { logger.LogDebug("No active recurrence rules found"); return; }

    var today = DateOnly.FromDateTime(DateTime.UtcNow);
    var horizon = today.AddDays(options.Value.GenerationHorizonDays);

    logger.LogInformation("Processing {RuleCount} active recurrence rules for horizon {From} to {To}",
        rules.Count, today, horizon);

    foreach (var rule in rules)
    {
        try { await ProcessRuleAsync(rule, today, horizon, bookingTypeRepo, tenantRepo, customerRepo, bookingRepo, unitOfWork, publisher, ct); }
        catch (Exception ex) when (ex is not OperationCanceledException)
        { logger.LogError(ex, "Error processing recurrence rule {RuleId}", rule.Id); }
    }
}

private async Task ProcessRuleAsync(
    RecurrenceRule rule, DateOnly today, DateOnly horizon,
    IBookingTypeRepository bookingTypeRepo, ITenantRepository tenantRepo,
    ICustomerRepository customerRepo, IBookingRepository bookingRepo,
    IUnitOfWork unitOfWork, IPublisher publisher, CancellationToken ct)
{
    var bookingType = await bookingTypeRepo.GetByIdAcrossTenantsAsync(rule.BookingTypeId, ct);
    if (bookingType is null)
    {
        logger.LogWarning("BookingType {BookingTypeId} not found for rule {RuleId} — skipping", rule.BookingTypeId, rule.Id);
        return;
    }

    var tenant = await tenantRepo.GetByIdAsync(rule.TenantId, ct);
    if (tenant is null)
    {
        logger.LogWarning("Tenant {TenantId} not found for rule {RuleId} — skipping", rule.TenantId, rule.Id);
        return;
    }

    var customer = await customerRepo.GetByIdAcrossTenantsAsync(rule.CustomerId, ct);
    if (customer is null)
    {
        logger.LogWarning("Customer {CustomerId} not found for rule {RuleId} — skipping", rule.CustomerId, rule.Id);
        return;
    }

    var tz = tenant.GetTimeZone();
    var occurrences = rule.ComputeOccurrences(today, horizon);

    BookingStatus[] conflictStatuses = [BookingStatus.PendingPayment, BookingStatus.PendingVerification, BookingStatus.Confirmed];

    foreach (var date in occurrences)
    {
        var localStart = new DateTime(date.Year, date.Month, date.Day, rule.StartTime.Hour, rule.StartTime.Minute, rule.StartTime.Second);
        var start = new DateTimeOffset(localStart, tz.GetUtcOffset(localStart));
        var end = start + rule.Duration;
        var (effStart, effEnd) = bookingType.GetEffectiveRange(start, end);

        var conflictCount = await bookingRepo.CountConflictsAsync(bookingType.Id, effStart, effEnd, conflictStatuses, ct);
        if (conflictCount >= bookingType.Capacity)
        {
            logger.LogDebug("Occurrence {Date} for rule {RuleId} skipped — slot full", date, rule.Id);
            continue;
        }

        var booking = Booking.Create(
            rule.TenantId, rule.BookingTypeId, start, end,
            rule.CustomerId.ToString(), customer.Email,
            bookingType.PriceInCentavos, bookingType.Currency);

        await bookingRepo.AddAsync(booking, ct);
        await unitOfWork.SaveChangesAsync(ct);

        await publisher.Publish(new BookingStatusChangedNotification(
            BookingId: booking.Id, TenantId: booking.TenantId,
            BookingTypeId: booking.BookingTypeId, BookingTypeSlug: string.Empty,
            FromStatus: null, ToStatus: booking.Status,
            Start: booking.Start, End: booking.End,
            CustomerId: booking.CustomerId, CustomerEmail: booking.CustomerEmail), ct);

        logger.LogInformation("Created recurring booking {BookingId} for rule {RuleId} on {Date}",
            booking.Id, rule.Id, date);
    }

    // Deactivate the rule if SeriesEnd has passed or we've hit MaxOccurrences.
    // MaxOccurrences check: ComputeOccurrences already caps output, so if we've
    // reached the horizon past SeriesEnd or the rule itself says it's exhausted, deactivate.
    if (rule.SeriesEnd.HasValue && rule.SeriesEnd.Value <= today)
    {
        rule.SoftDelete();
        // The repository needs to persist this change — use UpdateAsync if available, or a dedicated method.
        // For now: rely on UnitOfWork tracking (the rule entity is tracked if loaded via EF context).
        await unitOfWork.SaveChangesAsync(ct);
        logger.LogInformation("Recurrence rule {RuleId} deactivated — series ended", rule.Id);
    }
}
```

Note: `TenantTimeZone.GetUtcOffset(DateTime)` — check the actual API on `TenantTimeZone`. The existing `CreateBookingHandler` calls `tenant.GetTimeZone()` which returns `TenantTimeZone`, then `bookingType.ResolveSlot(cmd.StartTime, tz)`. Examine `TenantTimeZone` to understand how to convert a `DateOnly + TimeOnly` to `DateTimeOffset`.
